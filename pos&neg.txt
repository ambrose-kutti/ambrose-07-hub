#include<stdio.h>#include<stdlib.h>int main(){   int n;   scanf("%d",&n);   int i,j,k=0;   int arr[n],result[n/2];   for(i=0;i<n;i++)   {     scanf("%d",&arr[i]);   }   for(i=0;i<n;i++)   {     for(j=i+1;j<n;j++)     {       if(abs(arr[i])==abs(arr[j]))       {         result[k++]=abs(arr[i]);          }            }   }   for(i=0;i<k;i++)   {     printf("%d -%d ",result[i],result[i]);   }}
